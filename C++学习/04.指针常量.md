

```c++
int main() {
    int *s1, s2;

    //指针    地址 和 值 都可修改
    int *n1;
    n1 = s1;
    *n1 = s2;

    //指针常量  地址不能修改， 值可以修改
    int *const n2 = reinterpret_cast<int *const>(1);
    // n2 = s1;
    *n2 = s2;

    //常量指针  地址可以修改， 值不能修改
    const int *n3;
    n3 = s1;
    //*n3 = s2 ;

    //常量指针常量    地址 和 值 都不能修改
    const int *const n4 = nullptr;
    //n4 = s1;
    //*n4 = s2;

    return 0;
}
```
